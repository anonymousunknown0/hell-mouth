function playClick() {

function openFile(file) {

  playClick();

  document.getElementById('puzzle-grid').style.display = "none";

  const content = document.getElementById('content');

  const area = document.getElementById('content-area');

  const namebox = document.getElementById('file-name');

  let ext = file.split('.').pop().toLowerCase();

  area.innerHTML = "";

  namebox.innerHTML = file;

  if (ext === 'mp3' || ext === 'wav' || ext === 'm4a') {

    area.innerHTML = `<audio controls autoplay><source src="${file}"></audio>`;

  } else {

    area.innerHTML = `<video controls autoplay><source src="${file}"></video>`;

  }

  content.style.display = "flex";

  content.scrollIntoView({behavior: "smooth"});

}

function closeContent() {
